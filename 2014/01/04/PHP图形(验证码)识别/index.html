PHP图形(验证码)识别

发表于 2014-01-04 | 分类于 技术之道

因为一个项目的需要，需要验证码的识别，以前并没有接触过，后来研究了一下，做了个简单的PHP脚本(文章中的仅是做了简单整理,不是很严谨)，基本上满足了现在的需要，来分享一下，适合于基础选手。

验证码是这样的没有背景色，并且只有在深色的背景下才能看清，一共只有4个数字。

看了部分资料知道了思路，就很好办了。

1、图形按坐标转换为0,1

2、去除干扰(这个图片里没有)

3、切割

4、对比(主要是字典强大,用到的主要函数就是similar_text)

有了这四步就可以开工了。

//获取图片

public function getImage($url){

    if(!$url)return false;

    $dot = explode(&quot;.&quot;,$url);

    if($dot[count($dot)-1]==&quot;gif&quot;){

        $img = imagecreatefromgif($url);

    }

    else if($dot[count($dot)-1]==&quot;png&quot;){

        $img = imagecreatefrompng($url);

    }

    else if($dot[count($dot)-1]==&quot;jpg&quot;){

        $img = imagecreatefromjpeg($url);

    }else{

        $img  = false;

    }

    return $img;

}

//转为01

public function imgTobin($image){

    $data = array();

    for($y = 0; $y &lt; imagesy($image); $y++)

    {

        for($x = 0; $x &lt; imagesx($image); $x++)

        {

            $colors_reg = imagecolorsforindex($image, imagecolorat($image, $x, $y));

            if( $this-&gt;hexarray[ $colors_reg['red'] ].$this-&gt;hexarray[ $colors_reg['green'] ].$this-&gt;hexarray[ $colors_reg['blue'] ]=='abacad'){

                $data[$y][$x]=0;

            }else{

                if($this-&gt;hexarray[ $colors_reg['red'] ].$this-&gt;hexarray[ $colors_reg['green'] ].$this-&gt;hexarray[ $colors_reg['blue'] ]&gt;&quot;969696&quot;){

                    $data[$y][$x]=1;

                }else{

                    $data[$y][$x]=0;

                }

            }


        }

    }

    return $data;

}

public function getArea($data){

    foreach($data as  $lkey =&gt; $line){


        foreach($line as $fkey =&gt; $filed){

            if($line[$fkey-1]!=$line[$fkey]){

                $sprx[] = &quot;$fkey&quot;;

                $spry[] = &quot;$lkey&quot;;


            }

        }

    }


    sort($sprx);

    sort($spry);

    $sqare['min']['x'] = $sprx[0];

    $sqare['min']['y'] = $spry[0];

    $sqare['max']['x'] = $sprx[count($sprx)-1];

    $sqare['max']['y'] = $spry[count($spry)-1];

    $this-&gt;pwidth = ($sqare['max']['x']-$sqare['min']['x'])/4;

    $xx = 0;

    $yy = 0;

    foreach($data as  $lkey =&gt; $line){

        foreach($line as $fkey =&gt; $filed){

            if($fkey&gt;=$sqare['min']['x']&amp;&amp;$lkey&gt;=$sqare['min']['y']&amp;&amp;$fkey&lt;=$sqare['max']['x']&amp;&amp;$lkey&lt;=$sqare['max']['y']){

                $ndata[$yy][$xx]=$line[$fkey];

                $xx++;

            }

        }

        if($lkey&gt;=$sqare['min']['y']&amp;&amp;$lkey&lt;=$sqare['max']['y']){

            $yy++;

        }

    }

    return $ndata;

}

public function pExplode($ndata){

    for($i=0;$i&lt;4;$i++){

        $x = 0;

                    $word[$i].= $line[$fkey];

                }

                $x++;

            }

            $x = 0;

        }

    }

    return $word;

}

public function getNum($word){

    {

        $num = 0;

        foreach($this-&gt;keyvalues as $value)

        {

            $percent=0.0;

            $n[$i]['percent'] = intval($percent);

            $n[$i]['key'] = $value[0];

            $i++;

        }

        $percent = 0;

        foreach($n as $v){

            if($v['percent']&gt;$percent){

                $percent = $v['percent'];

                $num = $v['key'];

            }

        }

        unset($n);

        $result.=$num;

    }

    return  $result;

}
